oj %>% 
  ggplot(aes(x = log(sales), 
             y = log(price),
             color = brand)) +
  geom_point(alpha = .1) +
  geom_smooth(method = lm, se = F) +
  facet_wrap(.~ad) +
  theme_bw() +
  theme(legend.position = 'top')
